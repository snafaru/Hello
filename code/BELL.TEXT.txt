PROGRAM BELL;

CONST
        HOMECLR=12;

VAR
        INCHAR : CHAR;
        
PROCEDURE CURSBELL( CURSSTR : STRING); (* WIZARDRY CODE *)

  VAR
        X : INTEGER;

  BEGIN
    FOR X := 1 TO LENGTH( CURSSTR) DO
      BEGIN
        WRITE ( CURSSTR[ X]);
        WRITE ( CHR( 7));
        WRITE ( CHR( 7))
      END;
  END; (* CURSBELL *)

BEGIN (* MAIN PROGRAM *)

  WRITE( CHR( HOMECLR));
  
  GOTOXY( 0, 3);
  
  WRITELN( 'HELLO UCSD PASCAL!');
  WRITELN( '');
  WRITELN( 'LET US RING THE BELL THREE DIFFERENT WAYS IN WIZARDRY!');
  WRITELN( '');
  
  (* CLASSIC ASCII METHOD *)
  WRITE( 'PRESS A KEY TO RING THE BELL 3 TIMES USING THE CHR(7) METHOD! ');
  READ( INCHAR);
  WRITELN( '');
  WRITELN( 'A CRITICAL HIT!', CHR(7), CHR(7), CHR(7));
  (* WRITE( CHR(7), CHR(7), CHR(7)); *)
  WRITELN( '');
  
  (* METHOD AS PER POST BY THOMAS W. EWERS *)
  WRITE( 'PRESS TO RING THE BELL 3 TIMES WITH THE CTRL-G REPLACE METHOD! ');
  READ( INCHAR);
  WRITELN( '');
  WRITELN( 'A CRITICAL HIT!');
  (* WRITELN( 'T-O-M'); *)
  (* ON THE NEXT LINE THERE ARE 3 HIDDEN CTRL-G AFTER THE SEMICOLON *)
  WRITELN( '');
  
  (* METHOD AS PER WIZARDRY RE-ENGINEERED PASCAL CODE *)
  WRITE( 'PRESS A KEY TO RING THE BELL USING THE CURSBELL METHOD! ');
  READ( INCHAR);
  WRITELN( '');
  CURSBELL( '** CURSED **');
  WRITELN( '');
  WRITELN( '');
  WRITE( 'PRESS ANY KEY TO EXIT . . . ');
  READ( INCHAR);

END.
